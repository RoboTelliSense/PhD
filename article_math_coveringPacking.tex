\begin{equation}
\begin{array}{lllllll}
\mathrm{packing:} &\max &\mathbf{c}^T\mathbf{x}, & \ \ \ \ \mathrm{\ subject \ to\ } \mathbf{Ax}                    &\leq \mathbf{b}, &\mathbf{x} \geq 0 \\
\mathrm{covering:} &\min &\mathbf{b}^T\mathbf{y}, & \ \ \ \ \mathrm{\ subject \ to\ } \mathbf{A}^T\mathbf{y} &\geq \mathbf{c}, &\mathbf{y} \geq 0 \\
\end{array}
\end{equation}
