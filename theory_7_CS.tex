\section{Complexity}
%####################################################################################################

\begin{frame}
\frametitle{Complexity}
\framesubtitle{Big O notation}
\logoEvolution\mypagenum	
	\begin{block}{}
		\begin{align*}
			 f(x) &= O(g(x)) \ \text{as} \ x \rightarrow \infty \\
			\text{if and only if} \\
			|f(x)| &\leq  M|g(x) | \ \text{for all} \ x >x_0 \\
		\end{align*}
	\end{block}
	For example,
	\begin{align*}
		f(x) 				&= 6x^4-2x^3+5		\\
		|f(x)| 			&= 6x^4-|2x^3|+5	\\
						& \leq 6x^4-2x^4+5x^4\\
						&= M(x^4)			\\
		\Rightarrow f(x)	&=O(x^4)
	\end{align*}
\end{frame}
